\section*{Collaboration Methodology}

\subsection*{Version Control Strategy}
\begin{itemize}[leftmargin=*,itemsep=3pt]
  \item \textbf{Repository:} \url{https://github.com/vhakHCMUS/AWAD-BUS-PROJECT}
  \item \textbf{Branching Model:} Feature branches merged into \texttt{main} after code review
  \item \textbf{Main Branch:} \texttt{main} (production-ready code)
  \item \textbf{Sprint Branch:} \texttt{weekly-sprint} for milestone tracking
  \item \textbf{Commit Convention:} \texttt{FEAT:}, \texttt{FIX:}, \texttt{feat:}, \texttt{fix:} prefixes
  \item \textbf{Code Reviews:} Team lead (Khoa) reviewed all major feature implementations
\end{itemize}

\subsection*{Communication \& Coordination}
\begin{itemize}[leftmargin=*,itemsep=3pt]
  \item \textbf{Daily Standups:} Brief progress updates via messaging platform
  \item \textbf{Weekly Sprints:} Feature planning and milestone reviews
  \item \textbf{Documentation:} Comprehensive README, WIRING\_GUIDE, API docs (Swagger)
  \item \textbf{Issue Tracking:} GitHub issues for bug reports and feature requests
  \item \textbf{Code Standards:} Consistent formatting (Prettier for frontend, gofmt for backend)
\end{itemize}

\subsection*{Development Practices}
\begin{itemize}[leftmargin=*,itemsep=3pt]
  \item \textbf{Clean Architecture:} Layered design (handlers → usecases → services → repositories)
  \item \textbf{API Documentation:} Swagger/OpenAPI specification maintained throughout
  \item \textbf{Environment Management:} \texttt{.env} files for local, \texttt{.env.production} for deployment
  \item \textbf{Testing Scripts:} PowerShell scripts for testing booking, payment, chatbot flows
  \item \textbf{Continuous Deployment:} Automated deployment to Railway (backend) and Vercel (frontend)
\end{itemize}

\section*{Lessons Learned}

\subsection*{Technical Insights}
\begin{itemize}[leftmargin=*,itemsep=3pt]
  \item \textbf{Polling vs WebSocket:} For MVP, polling with aggressive caching provided sufficient "real-time" feel
  \item \textbf{Payment Security:} HMAC signature verification is critical but requires careful implementation
  \item \textbf{Dark Mode:} Implementing dark mode early would have saved refactoring time
  \item \textbf{Redis Caching:} Significantly improved performance for frequently accessed data (300s TTL for trips)
  \item \textbf{Clean Architecture:} Layered design made it easy to add features without breaking existing code
\end{itemize}

\subsection*{Collaboration Insights}
\begin{itemize}[leftmargin=*,itemsep=3pt]
  \item \textbf{Commit Messages:} Descriptive messages (FEAT/FIX prefixes) made history tracking easier
  \item \textbf{Feature Ownership:} Having clear owners for each feature reduced merge conflicts
  \item \textbf{Testing Scripts:} PowerShell test scripts enabled faster validation of API integrations
  \item \textbf{Documentation:} README and WIRING\_GUIDE helped onboard new features smoothly
  \item \textbf{Iterative Development:} Multiple small commits better than large monolithic changes
\end{itemize}
